[U-01] root 계정 원격 접속 제한
--------------------------------------------------

  cat /etc/securetty

[U-02] 패스워드 복잡도 설정

  cat /etc/pam.d/common-password
  cat /etc/pam.d/system-auth

[U-03] 계정 잠금 임계값 설정
--------------------------------------------------

  cat /etc/pam.d/system-auth

[U-04] 패스워드 파일 보호
--------------------------------------------------
  ls -al /etc/passwd
  ls -al /etc/shadow

[U-05] root 이외 UID 0 금지

  awk -F: '$3 == 0 { print $0 }' /etc/passwd

[U-06] root 계정 su 제한

  cat /etc/group | grep wheel

[U-07] 패스워드 최소 길이 설정

[D-01] DBMS 포트 외부 노출 여부
--------------------------------------------------
  # MySQL
  nc -zv <host> 3306
  # PostgreSQL
  nc -zv <host> 5432
  # MSSQL
  nc -zv <host> 1433

[D-02] 불필요 계정 제거 (기본 계정·빈 패스워드)
--------------------------------------------------
  # MySQL
  SELECT user, host, authentication_string FROM mysql.user;
  # PostgreSQL
  SELECT usename, passwd FROM pg_shadow;
  # MSSQL
  SELECT name, is_disabled FROM sys.server_principals WHERE type = 'S';

[D-03] 비밀번호 사용기간 및 복잡도 설정
--------------------------------------------------
  # MySQL
  SELECT @@version;
  SHOW VARIABLES LIKE 'validate_password%';
  # PostgreSQL
  SHOW password_encryption;
  # MSSQL
  SELECT name, is_policy_checked, is_expiration_checked FROM sys.sql_logins;

[D-04] 관리자 권한 최소화 (원격 root 로그인 제한)
--------------------------------------------------
  # MySQL
  SELECT user, host FROM mysql.user WHERE user='root' AND host != 'localhost';
  # PostgreSQL
  SELECT usename, usecreatedb, usesuper FROM pg_user;
  # MSSQL
  SELECT m.name FROM sys.server_role_members rm JOIN sys.server_principals m ON rm.member_principal_id=m.principal_id JOIN sys.server_principals r ON rm.role_principal_id=r.principal_id WHERE r.name='sysadmin';

[D-05] 감사 로깅 설정
--------------------------------------------------
  # MySQL
  SHOW VARIABLES LIKE 'general_log%';
  SHOW VARIABLES LIKE 'audit_log%';
  # PostgreSQL
  SHOW log_connections;
  SHOW log_disconnections;
  SHOW log_statement;
  # MSSQL
  SELECT * FROM sys.server_audits;

[D-06] 데이터 디렉토리 권한

  # MySQL
